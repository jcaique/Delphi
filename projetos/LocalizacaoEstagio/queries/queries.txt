create database LocalizacaoRegional

create table Pais (
	cd_Pais integer primary key identity(1,1),
	ds_Pais varchar(50)

	constraint un_pais unique(ds_Pais)
)

create table Estado (
cd_Estado integer primary key identity(1,1),
cd_Pais integer foreign key references Pais(cd_Pais),
ds_Estado varchar(50),
ds_Sigla varchar(2)

constraint un_estado unique(ds_Estado)
)

create table Municipio(
cd_Municipio integer primary key identity(1,1),
cd_Estado integer foreign key references Estado(cd_Estado),
ds_Municipio varchar(100)

constraint un_municipio unique(ds_municipio)
)
